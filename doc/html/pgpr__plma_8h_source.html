55  void cluster( const MatrixXd & test){
56  double sd = DBL_MAX;
57  double sq_dist = 0;
58  for(int testid = 0; testid < test.cols(); testid ++){
59  // Compute the local minimal distance
60  //The first point in the data is the centroid of each cluster
61  sd=0;
62  for(int j = 0 ; j < dim; j ++){
63  sd += SQR( localTrainSamples(j, 0) - test(j, testid));
64  }
65  sd = SQRT(sd);
66  struct{
67  double value;
68  int virMachineID;
69  } in, out;
70  in.value = sd;
71  in.virMachineID = myid;
72  MPI_Barrier(MPI_COMM_WORLD);
73  //use map-reduce method to locate the nearest cluster
74  MPI::COMM_WORLD.Allreduce(&in, &out, 1, MPI::DOUBLE_INT, MPI::MINLOC);
75  testClusters[out.virMachineID].push_back( test.col( testid));
76  }
